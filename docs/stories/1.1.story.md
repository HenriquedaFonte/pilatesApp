### Estrutura da Tabela `audit_log`
```sql
CREATE TABLE IF NOT EXISTS public.audit_log (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  actor_id     UUID NOT NULL,                    -- quem executou (auth.users.id)
  actor_email  TEXT NOT NULL,                    -- snapshot do email do executor
  action       TEXT NOT NULL,                    -- ex: 'student.delete', 'student.soft_delete'
  target_id    UUID,                             -- UUID do aluno afetado
  target_name  TEXT,                             -- snapshot do nome
  target_email TEXT,                             -- snapshot do email
  snapshot_data JSONB,                           -- dados completos do perfil no momento
  created_at   TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

### RLS da `audit_log`
```sql
-- Habilitar RLS
ALTER TABLE public.audit_log ENABLE ROW LEVEL SECURITY;

-- Apenas service role pode inserir
CREATE POLICY "service_role_insert_audit_log"
  ON public.audit_log FOR INSERT
  TO service_role
  WITH CHECK (true);

-- Nenhuma role pode UPDATE ou DELETE (append-only)
-- Leitura: apenas teacher logado
CREATE POLICY "teacher_select_audit_log"
  ON public.audit_log FOR SELECT
  USING (
    EXISTS (
      SELECT 1 FROM public.profiles
      WHERE id = auth.uid() AND role = 'teacher'
    )
  );
```

### ADD COLUMN em `profiles`
```sql
ALTER TABLE public.profiles
  ADD COLUMN IF NOT EXISTS deleted_at TIMESTAMPTZ,
  ADD COLUMN IF NOT EXISTS deleted_by UUID REFERENCES auth.users(id);
```

### Indexes recomendados
```sql
CREATE INDEX IF NOT EXISTS audit_log_target_id_idx ON public.audit_log(target_id);
CREATE INDEX IF NOT EXISTS audit_log_created_at_idx ON public.audit_log(created_at DESC);
CREATE INDEX IF NOT EXISTS profiles_deleted_at_idx ON public.profiles(deleted_at) WHERE deleted_at IS NOT NULL;
```
